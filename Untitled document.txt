from http.server import HTTPServer, SimpleHTTPRequestHandler
import os

# Set the directory you want to serve files from
directory_to_serve = "/path/to/your/directory"

# Change the current working directory to the directory to serve
os.chdir(directory_to_serve)

# Define the server's address and port
server_address = ('0.0.0.0', 8000)  # 0.0.0.0 allows access from other devices

# Start the HTTP server
httpd = HTTPServer(server_address, SimpleHTTPRequestHandler)

print(f"Serving files from {directory_to_serve} on http://{server_address[0]}:{server_address[1]}")
print("Accessible from other devices using your local IP address (e.g., http://192.168.x.x:8000)")

try:
    httpd.serve_forever()
except KeyboardInterrupt:
    print("\nShutting down the server.")
    httpd.server_close()
