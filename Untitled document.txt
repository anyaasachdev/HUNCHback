from http.server import HTTPServer, SimpleHTTPRequestHandler
import os

# Get the root directory of the file system
root_directory = "/"

# Change the working directory to the root of the filesystem
os.chdir(root_directory)

# Set up the server to listen on all network interfaces (not just localhost)
server_address = ('0.0.0.0', 8000)

# Create and start the HTTP server
httpd = HTTPServer(server_address, SimpleHTTPRequestHandler)
print(f"Serving all files from '{root_directory}' on {server_address[0]}:{server_address[1]}")
print("Accessible on the network via http://<your-ip>:8000")

try:
    httpd.serve_forever()
except KeyboardInterrupt:
    print("\nShutting down the server.")
    httpd.server_close()
